#include <iostream>
#include <iomanip>
#include <cstdlib>  // для system("cls") или system("clear")
using namespace std;

void clearScreen() {
    // Очистка экрана (кросс-платформенный способ)
    #ifdef _WIN32
        system("cls");
    #else
        system("clear");
    #endif
}

void printCentered(string text, int width = 50) {
    int padding = (width - text.length()) / 2;
    if (padding > 0) cout << string(padding, ' ');
    cout << text << endl;
}

void drawPlus(int size, char texture) {
    // Реализация для фигуры "Плюс"
    cout << "[ + ] Фигура: Плюс" << endl;
    cout << "[ + ] Размер: " << size << endl;
    cout << "[ + ] Текстура: " << texture << endl;
    cout << "[ + ] Результат:" << endl << endl;
    
    // Здесь должна быть логика рисования плюса
    // Пока просто заглушка
    cout << "    Фигура 'Плюс' будет нарисована здесь" << endl;
}

void drawRhombus(int size, char texture) {
    clearScreen();
    
    // Вывод шапки программы
    printCentered("[ 19 ] Самостоятельная работа | M");
    printCentered("Тема: [ FOR ]");
    printCentered("Продолжение программы \"Геометрические фигуры\".");
    cout << endl;
    printCentered("[ + ] Программа - \"Геометрические фигуры\"");
    cout << string(50, '-') << endl << endl;
    
    // Информация о выбранной фигуре
    cout << "[ + ] Фигура: \"Ромб\"" << endl << endl;
    cout << "[ + ] Размер: \t" << size << endl;
    cout << "[ + ] Текстура: \t" << texture << endl << endl;
    cout << "[ + ] Результат:" << endl << endl;
    
    // Проверка на корректность размера
    if (size % 2 == 0) {
        cout << "Ошибка: Для ромба нужен нечетный размер!" << endl;
        cout << "Используйте нечетное число (1, 3, 5, 7, ...)" << endl;
        return;
    }
    
    // Вычисляем середину
    int middle = size / 2;
    
    // Верхняя половина ромба (включая середину)
    for (int row = 0; row <= middle; row++) {
        for (int col = 0; col < size; col++) {
            // Определяем позиции для отрисовки символов
            if (col == middle - row || col == middle + row) {
                cout << texture << " ";
            } else {
                cout << ". ";
            }
        }
        cout << endl;
    }
    
    // Нижняя половина ромба (без середины)
    for (int row = middle - 1; row >= 0; row--) {
        for (int col = 0; col < size; col++) {
            if (col == middle - row || col == middle + row) {
                cout << texture << " ";
            } else {
                cout << ". ";
            }
        }
        cout << endl;
    }
}

int main() {
    clearScreen();
    
    // Шапка программы
    printCentered("[ 19 ] Самостоятельная работа | M");
    printCentered("Тема: [ FOR ]");
    printCentered("Продолжение программы \"Геометрические фигуры\".");
    cout << endl;
    printCentered("[ + ] Программа - \"Геометрические фигуры\"");
    cout << string(50, '-') << endl << endl;
    
    // Меню выбора фигур
    cout << "Доступные фигуры:" << endl;
    for (int i = 1; i <= 6; i++) {
        cout << "[ " << i << " ] Фигура " << i << endl;
    }
    cout << "[ 7 ] Плюс" << endl;
    cout << "[ 8 ] Ромб" << endl;
    cout << "[ 0 ] Выход" << endl << endl;
    
    int choice;
    cout << "[ + ] Выберите фигуру: ";
    cin >> choice;
    
    // Если выбран ромб
    if (choice == 8) {
        int size;
        char texture;
        
        cout << "[ + ] Размер: \t";
        cin >> size;
        
        cout << "[ + ] Текстура: \t";
        cin >> texture;
        
        drawRhombus(size, texture);
    }
    // Если выбран плюс
    else if (choice == 7) {
        int size;
        char texture;
        
        cout << "[ + ] Размер: \t";
        cin >> size;
        
        cout << "[ + ] Текстура: \t";
        cin >> texture;
        
        drawPlus(size, texture);
    }
    else if (choice == 0) {
        cout << "Выход из программы..." << endl;
    }
    else {
        cout << "Фигура №" << choice << " еще не реализована." << endl;
    }
    
    return 0;
}
